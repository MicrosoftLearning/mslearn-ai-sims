Store original column names for user-friendly input original_columns = list(X.columns) log_message(f" • Original feature columns: {original_columns}") # Identify categorical and numeric columns for pipeline log_message("📋 Analyzing column types for pipeline...") categorical_cols = [] numeric_cols = [] columns_to_ignore = [] for col in X.columns: if col in categorical_settings: if categorical_settings[col] == 'ignore': columns_to_ignore.append(col) continue elif categorical_settings[col] == 'categorize': if X[col].dtype == 'object': categorical_cols.append(col) else: numeric_cols.append(col) else: #
